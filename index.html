const GITHUB_TOKEN = "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"; // Replace with your token const REPO_OWNER = "CRAZFREE"; // Replace with your GitHub username const REPO_NAME = "anonymous"; // Replace with your repository name const FILE_PATH = "messages.json"; // Path to the file in the repo async function submitMessage(message) { try { // Step 1: Fetch the existing file const response = await fetch( `https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${FILE_PATH}`, { headers: { Authorization: `Bearer ${GITHUB_TOKEN}`, }, } ); if (!response.ok) { throw new Error("Failed to fetch existing file"); } const fileData = await response.json(); const currentContent = JSON.parse(atob(fileData.content)); // Step 2: Update the content currentContent.push({ message, timestamp: new Date().toISOString() }); const updatedContent = btoa(JSON.stringify(currentContent, null, 2)); // Step 3: Push the update const updateResponse = await fetch( `https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${FILE_PATH}`, { method: "PUT", headers: { Authorization: `Bearer ${GITHUB_TOKEN}`, "Content-Type": "application/json", }, body: JSON.stringify({ message: "Add new message", content: updatedContent, sha: fileData.sha, // Required for updating the file }), } ); if (updateResponse.ok) { alert("Message submitted successfully!"); } else { throw new Error("Failed to update file"); } } catch (error) { console.error(error); alert("Failed to submit the message."); } }
